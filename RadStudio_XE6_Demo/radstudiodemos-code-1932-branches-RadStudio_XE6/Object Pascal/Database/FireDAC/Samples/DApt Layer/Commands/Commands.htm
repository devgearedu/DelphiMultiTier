1         Commands

The demo demonstrates using standalone table adapter to redirect inserting, deleting and updating records.

  1. Create table adapter

ADDAptManager.CreateTableAdapter(oAdapt);

  1. Selecting data

with oAdapt do begin

  FConnIntf.CreateCommand(oComm);

  SelectCommand := oComm;

  SelectCommand.Prepare('select * from {id ADQA_map1}');

  Define;

  Fetch;

end;

  1. Redirect all record inserts into ADQA_map2 table instead ADQA_map1

with oAdapt do begin  

  FConnIntf.CreateCommand(oComm);

  InsertCommand := oComm;

  InsertCommand.CommandText := 'insert into {id ADQA_map2}(id2, name2) ‘ +

    ‘values(:NEW_id1, :NEW_name1)';

end;

  1. Redirect all record deletes into ADQA_map3 table instead ADQA_map1

with oAdapt do begin

  FConnIntf.CreateCommand(oComm);

  DeleteCommand := oComm;

  DeleteCommand.CommandText := 'delete from {id ADQA_map3} where id3 = :OLD_id1';

end;

  1. Redirect all record updates into ADQA_map4 table instead ADQA_map1

with oAdapt do begin

  FConnIntf.CreateCommand(oComm);

  UpdateCommand := oComm;

  UpdateCommand.CommandText := 'update {id ADQA_map4} ‘ +

    ‘set id4 = :NEW_id1, name4 = :NEW_name1 where id4 = :OLD_id1';

end;

  1. Post data to RDBMS

oAdapt.Update;
